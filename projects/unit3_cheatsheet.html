Commonly used methods
  1. .include

    2. arr = ['a', 'b', 'c']
    arr.include?('a') #=> true
    arr.include?('y') #=> false

  2. .gsub

    3. user_input = "hello there"
    if user_input.include? "there"
     user_input.gsub!("there", "you")
    end #=> "hello you"

  3. .inject

    4. [1,2,3,4].inject(0){|result, element| result + element}
    1 + 2 + 3 + 4 #=> 10

  4. .flatten

    5. a = [1, 2, 3]
    b = [w, x, y]
    c = [a, b, 9, 10] #=> [[1, 2, 3], [w, x, y], 9, 10]
    c.flatten #=> [1, 2, 3, w, x, y, 9, 10]

Hashes and Arrays

  1. Hash
    • How to create a new hash
      1. apple = Hash.new #=> apple = {}
      2. drinks = Hash.new {|hash, key| hash[key] = "oranges: #{key}"} #=> {}
        drinks['screwdriver'] #=> "oranges: screwdriver"
        drinks #=> {"screwdriver"=>"oranges: screwdriver"}
      3. cars = {
           "Toyota" => "Camry",
           "Honda" => "Civic",
           "Ford" => "Taurus"
        }
        #=> {"Toyota":"Camry","Honda":"Civic", "Ford":"Taurus"}
    • How to assign a new value for an element of a hash

      • cars = {
         "Toyota" => "Camry",
         "Honda" => "Civic",
         "Ford" => "Taurus"
      }
      cars["Toyota"] = "Corolla"
      #=> "Corolla"
      cars #=> {"Toyota"=>"Corolla", "Honda"=>"Civic", "Ford"=>"Taurus"}

Iteration

  1. If

    2. def glass_color(color)
       if glass == "Red"
          puts "The glass is red."
       elsif glass == "Green"
          puts "The glass is green."
       else
          puts "The glass is not red or green."
       end
    end
    glass_color("Blue") #=> "The glass is not red or green."

    def color(color)
       color == "Red"? "It's red": color == "Green"? "It's green":"It's not red or green."
    end color("Black") #=> "It's not red or green."

  2. Loop do

    3. Don't forget the counter or else it will go on forever!
    counter = 0
    loop do
       print "woot "
       counter += 1
       break if counter = 5
    end #=> woot woot woot woot woot

  3. For

    4. for number in 1..5
       puts number
    end

  4. Each

    5. object.each{|item| #do something}
    numbers = [1,2,3,4,5]
    numbers.each{|num| puts num} #=> 1 2 3 4 5

    object.each do |item|
       do something
    end
    numbers = [1,2,3,4,5]
    numbers.each do |num|
       puts num
    end #=> 1 2 3 4 5

  5. Unless

    6. bored = false until bored    puts "I'm writing Ruby programs!" end #=> I'm writing Ruby programs!"

  6. Times

    7. 5.times{puts "beep"} #=> beep beep beep beep beep

  7. Until

    8. Don't forget about the counter
    counter = 3
    until counter == 0 do
       puts "boom"
       counter -= 1
    end #=> boomboomboom

  8. Case (When)

    9. puts "What language do you speak?"
    greeting = gets.chomp
    case greeting
    when "English" then puts "Hello!"
    when "French" then puts "Bonjour!"
    when ...
    else
    puts "I don't know that language."
    end
